# RunPod XTTS-mini one-shot setup (RTX 3090 / CUDA 12.8 / PyTorch cu128)
# Designed specifically for xtts_mini.py

mkdir -p /workspace/xtts_job && cd /workspace/xtts_job

python3 -m venv venv
source venv/bin/activate

pip install -U pip setuptools wheel

# ------------------------------------------------------------------------------
# PyTorch / CUDA stack
# IMPORTANT:
# - Keep torch + torchaudio on matching versions
# - This pod is working with CUDA 12.8, so use cu128 wheels
# ------------------------------------------------------------------------------
pip install \
  torch==2.7.0 \
  torchvision==0.22.0 \
  torchaudio==2.7.0 \
  --index-url https://download.pytorch.org/whl/cu128

# ------------------------------------------------------------------------------
# XTTS / Coqui compatibility pins
# These keep XTTS stable and avoid newer breaking dependency drift
# ------------------------------------------------------------------------------
pip install \
  "TTS==0.22.0" \
  "transformers==4.38.2" \
  "tokenizers==0.15.2" \
  "safetensors>=0.4.2" \
  "huggingface_hub>=0.20,<1.0" \
  "soundfile>=0.12.1" \
  "numpy>=1.26,<2.0"

required = [
    "/workspace/xtts_job/xtts_v2_model/config.json",
    "/workspace/xtts_job/xtts_v2_model/model.pth",
    "/workspace/xtts_job/xtts_v2_model/vocab.json",
